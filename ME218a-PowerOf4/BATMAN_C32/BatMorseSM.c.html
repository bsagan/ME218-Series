  132  Function
  133     RunTemplateFSM
  134 
  135  Parameters
  136    ES_Event : the event to process
  137 
  138  Returns
  139    ES_Event, ES_NO_EVENT if no error ES_ERROR otherwise
  140 
  141  Description
  142    add your description here
  143  Notes
  144    uses nested switch/case to implement the machine.
  145  Author
  146    J. Edward Carryer, 01/15/12, 15:23
  147 ****************************************************************************/
  148 ES_Event RunBatMorseSM( ES_Event ThisEvent )
  149 {
  150     static MorseState_t NextState;
  151 
  152     ES_Event ReturnEvent;
  153     ReturnEvent.EventType = ES_NO_EVENT; // assume no errors
  154 
  155     CurrentState = QueryBatMorseSM();
  156     //NextState = CurrentState;
  157 
  158     switch (CurrentState)
  159         {
  160         case InitMorseElems:
  161             if (ThisEvent.EventType == ES_INIT)
  162                 {
  163                     NextState = CalWaitForRise;
  164                     printf("CalWaitRise\r\n");
  165                 }
  166             break;
  167 
  168         case CalWaitForRise:
  169             switch (ThisEvent.EventType)
  170                 {
  171                 case RISING_EDGE:
  172                 {
  173                     TimeofLastRise = ThisEvent.EventParam;
  174                     NextState = CalWaitForFall;
  175                     printf("CalWaitFall\r\n");
  176                 }
  177                 break;
  178                 case CAL_COMPLETE:
  179                 {
  180                     NextState = EOC_WaitRise;
  181                     printf("EOCWaitRise\r\n");
  182                 }
  183                 break;
  184                 }
  185             break;
  186 
  187         case CalWaitForFall:
  188             if (ThisEvent.EventType == FALLING_EDGE)
  189                 {
  190                     TimeofLastFall = ThisEvent.EventParam;
  191                     NextState = CalWaitForRise;
  192                     printf("CalWaitRise\r\n");
  193 
  194                     TestCalibration();
  195                 }
  196             break;
  197 
  198         case EOC_WaitRise:
  199             switch (ThisEvent.EventType)
  200                 {
  201                 case RISING_EDGE:
  202                 {
  203                     TimeofLastRise = ThisEvent.EventParam;
  204                     NextState = EOC_WaitFall;
  205                     printf("EOCWaitFall\r\n");
  206                     CharacterizeSpace();
  207                 }
  208                 break;
  209                 }
  210             break;
  211 
  212 
  213         case EOC_WaitFall:
  214             switch (ThisEvent.EventType)
  215                 {
  216                 case FALLING_EDGE:
  217                 {
  218                     TimeofLastFall = ThisEvent.EventParam;
  219                     NextState = EOC_WaitRise;
  220                     printf("EOCWaitRise\r\n");
  221                 }
  222                 break;
  223                 case EOC_Detected:
  224                 {
  225                     NextState = DecodeWaitFall;
  226                     printf("decodewaitfall\r\n");
  227                 }
  228                 break;
  229                 }
  230             break;
  231 
  232         case DecodeWaitRise:
  233             switch (ThisEvent.EventType)
  234                 {
  235                 case RISING_EDGE:
  236                 {
  237                     TimeofLastRise = ThisEvent.EventParam;
  238                     NextState = DecodeWaitFall;
  239                     //printf("decodewaitfall\r\n");
  240 
  241                     CharacterizeSpace();
  242                 }
  243                 break;
  244                 }
  245             break;
  246 
  247 
  248         case DecodeWaitFall:
  249             switch (ThisEvent.EventType)
  250                 {
  251                 case FALLING_EDGE:
  252                 {
  253                     TimeofLastFall = ThisEvent.EventParam;
  254                     NextState = DecodeWaitRise;
  255                     //printf("decodewaitrise\n");
  256                     CharacterizePulse();
  257                 }
  258                 break;
  259                 }
  260             break;
  261         }
  262     CurrentState = NextState;
  263 
  264     return ReturnEvent;
  265 }

  345 ////////////////////CHARACTERIZE SPACE
  346 void CharacterizeSpace(void)
  347 {
  348     static uint16_t LastInterval;
  349 
  350     ES_Event ThisEvent;
  351 
  352 
  353     LastInterval = TimeofLastRise - TimeofLastFall;
  354 //printf("last space interval %i\r\n", LastInterval);
  355     if ((LastInterval >= (2*LengthOfDot - 10)) && (LastInterval  <= (2*LengthOfDot + 15)))
  356         {
  357 
  358             ThisEvent.EventType = EOC_Detected;
  359             ThisEvent.EventParam = 1;
  360             PostBatMorseSM(ThisEvent);
  361             PostBatDecodeSM(ThisEvent);
  362             //printf("EOC_detected with interval %i\r\n", LastInterval);
  363 
  364         }
  365 
  366     else
  367         {
  368             if ((LastInterval >= (6*LengthOfDot - 15)) && (LastInterval <= (6*LengthOfDot + 15)))
  369                 {
  370                     ThisEvent.EventType = EOW_Detected;
  371                     ThisEvent.EventParam = 1;
  372                     PostBatMorseSM(ThisEvent);
  373                     PostBatDecodeSM(ThisEvent);
  374                     //printf("EOw_detected with interval %i\r\n", LastInterval);
  375                 }
  376             else
  377                 {
  378                     ThisEvent.EventType = BAD_SPACE;
  379                     ThisEvent.EventParam = 1;
  380                     PostBatMorseSM(ThisEvent);
  381                     PostBatDecodeSM(ThisEvent);
  382                 }
  383         }
  384     return;
  385 }

  389 ///////////////////CHARACTERIZE PULSE
  390 void CharacterizePulse(void)
  391 {
  392     static uint16_t LastPulseWidth;
  393 
  394     ES_Event ThisEvent;
  395 
  396 
  397     LastPulseWidth = TimeofLastFall - TimeofLastRise;
  398 //printf("last pulse width %i\r\n",LastPulseWidth);
  399     if ((LastPulseWidth <= (LengthOfDot + 5)) && (LastPulseWidth >= (LengthOfDot - 5)))
  400         {
  401 
  402             ThisEvent.EventType = DOT_Detected;
  403             ThisEvent.EventParam = 1;
  404             PostBatDecodeSM(ThisEvent);
  405             //printf("dot_detected with interval %i\r\n", LastPulseWidth);
  406 
  407 
  408         }
  409     else
  410         {
  411             if ((LastPulseWidth >=(3*LengthOfDot - 20)) && (LastPulseWidth <=(3*LengthOfDot + 10)))
  412                 {
  413                     ThisEvent.EventType = DASH_Detected;
  414                     ThisEvent.EventParam = 1;
  415                     PostBatDecodeSM(ThisEvent);
  416                     //printf("dash_detected with interval %i\r\n", LastPulseWidth);
  417                 }
  418             else
  419                 {
  420                     ThisEvent.EventType = BAD_PULSE;
  421                     ThisEvent.EventParam = 1;
  422                     PostBatDecodeSM(ThisEvent);
  423                 }
  424         }
  425     return;
