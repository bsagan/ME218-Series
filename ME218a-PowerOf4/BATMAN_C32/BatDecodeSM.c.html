  137  Function
  138     RunTemplateFSM
  139 
  140  Parameters
  141    ES_Event : the event to process
  142 
  143  Returns
  144    ES_Event, ES_NO_EVENT if no error ES_ERROR otherwise
  145 
  146  Description
  147    add your description here
  148  Notes
  149    uses nested switch/case to implement the machine.
  150  Author
  151    J. Edward Carryer, 01/15/12, 15:23
  152 ****************************************************************************/
  153 ES_Event RunBatDecodeSM( ES_Event ThisEvent )
  154 {
  155 
  156     static char Char2Display; //initialize output from DecodeMorse()
  157 
  158     ES_Event ReturnEvent;
  159     ReturnEvent.EventType = ES_NO_EVENT; // assume no errors
  160 
  161 
  162     switch (ThisEvent.EventType)
  163         {
  164         case DOT_Detected:
  165         {
  166 
  167             //max strlen = 7 b.c 8th entry should be null
  168             if (MorseIndex <=7 )   //strlen() determines size btw 1st and null terminating character
  169                 {
  170                     MorseString[MorseIndex] = '.';
  171                     //printf("Added dot\r\n");
  172                     MorseIndex++;
  173                     return ReturnEvent;
  174                 }
  175             else
  176                 {
  177                     puts("something went wrong in DecodeSM Dot_detection \r");
  178                     ClearMorseChar();
  179                     return ReturnEvent;
  180                 }
  181         }
  182         break;
  183 
  184         case DASH_Detected:
  185         {
  186 
  187             if (MorseIndex <=7 )   //strlen() determines size btw 1st and null terminating character
  188                 {
  189                     MorseString[MorseIndex] = '-';
  190                     //printf("Added dash\r\n");
  191                     MorseIndex++;
  192                     return ReturnEvent;
  193                 }
  194             else
  195                 {
  196                     puts("something went wrong in DecodeSM Dash_detection \r");
  197                     ClearMorseChar();
  198                     return ReturnEvent;
  199                 }
  200         }
  201         break;
  202 
  203         case EOC_Detected:
  204         {
  205             if (DecodeMorse() != '#')
  206                 {
  207 
  208                     Char2Display = DecodeMorse();
  209                     //printf("%c\r\n", Char2Display);
  210                     if (MsgIndex <=5)
  211                         {
  212                             MorseMsg[MsgIndex] = Char2Display;
  213                             MsgIndex++;
  214                             ClearMorseChar();
  215                         }
  216                     else
  217                         {
  218                             if (MatchMorse() == True)
  219                                 {
  220                                     ES_Event MorseEvent;
  221                                     printf("morse msg %s\r\n", MorseMsg);
  222                                     ThisEvent.EventType = BATID_OK;
  223                                     ThisEvent.EventParam = 1;
  224                                     PostGameplaySM(ThisEvent);
  225 
  226 
  227                                     MorseEvent.EventType = InitMorseElems;
  228                                     MorseEvent.EventParam = 0;
  229                                     PostBatMorseSM(MorseEvent);
  230 
  231 
  232                                     ClearMorseChar();
  233                                     ClearMorseMsg();
  234 
  235                                 }
  236                             ClearMorseChar();
  237                             ClearMorseMsg();
  238                         }
  239 
  240                 }
  241             else
  242                 {
  243                     //printf("morse string %s\r\n", MorseString);
  244                     //puts("decodemorse could not decode character \r");
  245                     ClearMorseChar();
  246                     return ReturnEvent;
  247                 }
  248         }
  249         break;
  250 
  251         case EOW_Detected:
  252         {
  253             if (DecodeMorse() != '#')
  254                 {
  255                     Char2Display = DecodeMorse();
  256                     ClearMorseChar();
  257                 }
  258             else
  259                 {
  260                     //puts("decodemorse could not decode last character in word \r");
  261                     ClearMorseChar();
  262                     return ReturnEvent;
  263                 }
  264         }
  265         break;
  266 
  267         }
  268     return ReturnEvent;
  269 }

  296 ////////////////////ClearMorseChar
  297 void ClearMorseChar(void)
  298 {
  299     unsigned int n; //loop counter
  300     for (n = 0; n<8; n++)
  301         {
  302             MorseString[n] = '\0';
  303         }
  304     //printf("morse string cleared %s\r\n", MorseString);
  305     MorseIndex = 0;
  306 
  307     return;
  308 }

  311 ///////////////////DecodeMorse
  312 char DecodeMorse(void)
  313 {
  314 
  315     unsigned int n; //loop counter
  316     unsigned int index;
  317     char Flag = '#';
  318     char legalChars[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890?.,:'-/()\"= !$&+;@_";
  319     char morseCode[][8] = { ".-","-...","-.-.","-..",".","..-.","--.",
  320                             "....","..",".---","-.-",".-..","--","-.","---",
  321                             ".--.","--.-",".-.","...","-","..-","...-",
  322                             ".--","-..-","-.--","--..",".----","..---",
  323                             "...--","....-",".....","-....","--...","---..",
  324                             "----.","-----","..--..",".-.-.-","--..--",
  325                             "---...",".----.","-....-","-..-.","-.--.-",
  326                             "-.--.-",".-..-.","-...-","-.-.--","...-..-",
  327                             ".-...",".-.-.","-.-.-.",".--.-.","..--.-"
  328                           };
  329     //printf("morse string %s with length %i \r\n", MorseString, strlen(MorseString));
  330 
  331     for (n = 0; n < 50; n++)
  332         {
  333             if (strcmp(morseCode[n], MorseString) == 0)
  334                 {
  335                     index = n;
  336                     return legalChars[n];
  337                 }
  338 
  339 
  340         }
  341     return Flag;
  342 
  343 }
