   51 /****************************************************************************
   52  Function
   53      InitPortSampleSM
   54 
   55  Parameters
   56      uint8_t : the priorty of this service
   57 
   58  Returns
   59      boolean, False if error in initialization, True otherwise
   60 
   61  Description
   62      Saves away the priority, sets up the initial transition and does any
   63      other required initialization for this state machine
   64  Notes
   65 
   66  Author
   67      J. Edward Carryer, 10/23/11, 18:55
   68 ****************************************************************************/
   69 boolean InitPortSampleSM ( uint8_t Priority )
   70 {
   71     ES_Event ThisEvent;
   72 
   73     ADS12_Init("AAIIIIII"); //initializes AD ports 6,7 as analog inputs, ports 0-5 as digital inputs
   74     if (ADS12_Init("AAIIIIII") == ADS12_OK)
   75         {
   76             printf("Analog library intialized okay \r\n");
   77         }
   78     MyPriority = Priority;
   79     // put us into the Initial PseudoState
   80     CurrentState = InitPStatePortSample;
   81     // post the initial transition event
   82     ThisEvent.EventType = ES_INIT;
   83     if (ES_PostToService( MyPriority, ThisEvent) == True)
   84         {
   85             return True;
   86         }
   87     else
   88         {
   89             return False;
   90         }
   91 }

  115 /****************************************************************************
  116  Function
  117     RunPortSampleSM
  118 
  119  Parameters
  120    ES_Event : the event to process
  121 
  122  Returns
  123    ES_Event, ES_NO_EVENT if no error ES_ERROR otherwise
  124 
  125  Description
  126    add your description here
  127  Notes
  128    uses nested switch/case to implement the machine.
  129  Author
  130    J. Edward Carryer, 01/15/12, 15:23
  131 ****************************************************************************/
  132 ES_Event RunPortSampleSM( ES_Event ThisEvent )
  133 {
  134     ES_Event ReturnEvent;
  135     ReturnEvent.EventType = ES_NO_EVENT; // assume no errors
  136 
  137     switch ( CurrentState )
  138         {
  139         case InitPStatePortSample :       // If current state is initial Psedudo State
  140             if ( ThisEvent.EventType == ES_INIT )// only respond to EF_Init
  141                 {
  142                     //PORT DIRECTIONS INITIALIZED IN AD INITIALIZATION
  143                     /* Initialize port directions
  144 
  145                     OIL_SLICK_DIR  = 0;
  146                     LASER_BEAM_DIR = 0;
  147                     SIDE_BUTTON_DIR = 0;
  148                     BOMB_BUTTON_DIR = 0;
  149                     BOTTOM_LIMIT_DIR = 0;
  150                     TOP_LIMIT_DIR = 0; */
  151 
  152                     PseudoPort[OIL_SLICK_POS] = OIL_SLICK_PORT;
  153                     PseudoPort[LASER_BEAM_POS] = LASER_BEAM_PORT;
  154                     PseudoPort[SIDE_BUTTON_POS] = SIDE_BUTTON_PORT;
  155                     PseudoPort[BOMB_BUTTON_POS] = BOMB_BUTTON_PORT;
  156                     PseudoPort[BOTTOM_LIMIT_POS] = BOTTOM_LIMIT_PORT;
  157                     PseudoPort[TOP_LIMIT_POS] = TOP_LIMIT_PORT;
  158                     PseudoPort[PEDAL_POS] = ADS12_ReadADPin(PEDAL_PORT);
  159 
  160                     ES_Timer_InitTimer(SAMPLE_TIMER, DEBOUNCE_TIME);
  161 
  162                     // now put the machine into the actual initial state
  163                     CurrentState = PseudoPortWriting;
  164 
  165                 }
  166             break;
  167 
  168         case PseudoPortWriting :       // If current state is PseudoPortWriting
  169             if (ThisEvent.EventType == ES_TIMEOUT && ThisEvent.EventParam == SAMPLE_TIMER)
  170                 {
  171 
  172                     PseudoPort[OIL_SLICK_POS] = OIL_SLICK_PORT;
  173                     PseudoPort[LASER_BEAM_POS] = LASER_BEAM_PORT;
  174                     PseudoPort[SIDE_BUTTON_POS] = SIDE_BUTTON_PORT;
  175                     PseudoPort[BOMB_BUTTON_POS] = BOMB_BUTTON_PORT;
  176                     PseudoPort[BOTTOM_LIMIT_POS] = BOTTOM_LIMIT_PORT;
  177                     PseudoPort[TOP_LIMIT_POS] = TOP_LIMIT_PORT;
  178                     PseudoPort[PEDAL_POS] = ADS12_ReadADPin(PEDAL_PORT);
  179 
  180                     ES_Timer_InitTimer(SAMPLE_TIMER, DEBOUNCE_TIME);
  181 
  182                     //printf("Pedal Value is, %i\r\n", PseudoPort[PEDAL_POS]);
  183 
  184                     // Ensure we stay in Writing State
  185                     CurrentState = PseudoPortWriting;
  186                 }
  187             break;
  188         }                                   // end switch on Current State
  189     return ReturnEvent;
  190 }
