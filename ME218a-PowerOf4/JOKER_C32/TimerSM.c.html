   63  Function
   64      InitTimerSM
   65 
   66  Parameters
   67      uint8_t : the priorty of this service
   68 
   69  Returns
   70      boolean, False if error in initialization, True otherwise
   71 
   72  Description
   73      Saves away the priority, sets up the initial transition and does any
   74      other required initialization for this state machine
   75  Notes
   76 
   77  Author
   78      J. Edward Carryer, 10/23/11, 18:55
   79 ****************************************************************************/
   80 boolean InitTimerSM ( uint8_t Priority )
   81 {
   82     ES_Event ThisEvent;
   83 
   84     puts("Timer has entered initialization\r");
   85 
   86     MyPriority = Priority;
   87     // put us into the Initial PseudoState
   88     CurrentState = InitPStateTimer;
   89     // post the initial transition event
   90     ThisEvent.EventType = ES_INIT;
   91     if (ES_PostToService( MyPriority, ThisEvent) == True)
   92         {
   93             return True;
   94         }
   95     else
   96         {
   97             return False;
   98         }
   99 }

  128    ES_Event : the event to process
  129

  133  Description

  140 ES_Event RunTimerSM( ES_Event ThisEvent )
  141 {
  142     ES_Event ReturnEvent;
  143     ReturnEvent.EventType = ES_NO_EVENT; // assume no errors
  144 
  145     switch ( CurrentState )
  146         {
  147         case InitPStateTimer :       // If current state is initial Psedudo State
  148             if ( ThisEvent.EventType == ES_INIT )// only respond to EF_Init
  149                 {
  150                     // this is where you would put any actions associated with the
  151                     // transition from the initial pseudo-state into the actual
  152                     // initial state
  153 
  154                     // now put the machine into the actual initial state
  155 
  156                     printf("I'm in InitPStateTimer\r\n");
  157 
  158                     CurrentState = TimerIdle;
  159 
  160                     printf("I have set CurrentState to TimerIdle\r\n");
  161 
  162                     return ReturnEvent;
  163                 }
  164             break;
  165 
  166         case TimerIdle :       // If current state is state one
  167 
  168             printf("I'm in TimerIdle\r\n");
  169 
  170             switch ( ThisEvent.EventType )
  171                 {
  172                 case GAMESTART : //If event is event one
  173                 {
  174                     // Initialize GameTimer
  175                     ES_Timer_InitTimer(GAME_TIMER, GAME_TIME);
  176 
  177                     // Call function to Light All Rows of Lights
  178                     printf("I should light 9 rows of lights\r\n");
  179                     TimerLights(Timer9ON);
  180                     TimeVal = TIME_COUNTS;
  181 
  182                     CurrentState = CountDown;//Decide what the next state will be
  183 
  184                 }
  185                 break;
  186                 }
  187             break;
  188 
  189         case CountDown :       // If current state is state one
  190 
  191             printf("I'm in CountDown\r\n");
  192 
  193             switch ( ThisEvent.EventType )
  194                 {
  195                 case ES_TIMEOUT : //If event is event one
  196                 {
  197                     if (ThisEvent.EventParam == GAME_TIMER)
  198                         {
  199                             CurrentState = CountDown;//Decide what the next state will be
  200 
  201                             // Call evaluate time function and determine how many light rows on
  202                             EvalTime();
  203                         }
  204                 }
  205                 break;
  206 
  207                 case BATMANWIN :
  208                 {
  209                     TimerLights(OFF);
  210                     CurrentState = TimerIdle;
  211                 }
  212                 break;
  213 
  214                 case BATMANLOSE :
  215                 {
  216                     TimerLights(OFF);
  217                     CurrentState = TimerIdle;
  218                 }
  219                 break;
  220 
  221                 case GAME_END :
  222                 {
  223                     ES_Event ThisEvent;
  224                     ThisEvent.EventType = GAME_END;
  225                     ThisEvent.EventParam = 0;
  226                     PostGameplaySM(ThisEvent);
  227                     CurrentState = TimerIdle;
  228                 }
  229                 break;
  230 
  231                 case DETONATE :
  232                 {
  233                     TimerLights(OFF);
  234                     CurrentState = TimerIdle;
  235                 }
  236                 break;
  237 
  238                 CurrentState = CountDown;
  239 
  240                 }
  241             break;
  242         }
  243     return ReturnEvent;
  244 }
