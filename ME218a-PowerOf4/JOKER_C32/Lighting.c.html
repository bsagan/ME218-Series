  126 void InitLights(void)
  127 {
  128     //initialize ports
  129     SER_DIR = 1; //set serial port as output
  130     SRCLK_DIR = 1; //set shift register clock as output
  131     SRCLR_DIR = 1; //set shift register clear as output
  132     RCLK_DIR = 1; //set storage clock as output
  133     SER = 0;
  134     pulseSRCLR();
  135 }
  136 
  137 void pulseSRCLK(void) //pulses SRCLK high
  138 {
  139     SRCLK = 1; //set port high
  140     SRCLK = 0;
  141 }
  142 
  143 void pulseRCLK(void) //pulses RCLK high
  144 {
  145     RCLK = 1; //set port high
  146     RCLK = 0;
  147 }
  148 
  149 void pulseSRCLR(void) //pulses SRCLR low
  150 {
  151     static int i;
  152     SRCLR = 0; //set port low
  153     for (i=0; i<2; i++)
  154         {
  155 
  156         }
  157     pulseRCLK(); //show bits on port
  158     SRCLR = 1;
  159 }

  161 void lightLEDs(uint32_t bitNums)
  162 //takes in a 16bit number with 0 indicating LED off, 1 on
  163 //MSB --> LSB reads QH --> QA
  164 //mask each bit and set SER high if 1 or SER low if 0
  165 //SER = bitNums & mask, pulse SRCLK, shift mask, repeat 8 times
  166 //after 8 shifts, pulse RCLK
  167 //shift mask right 1 bit: variable >> 1;
  168 
  169 {
  170     uint32_t mask = BIT31HI;
  171     unsigned char i;
  172     uint32_t blah;
  173 
  174     //printf("bitNums %lu\r\n",bitNums);
  175     pulseSRCLR();
  176     for (i=0; i<=31; i++) //loop thru 8 bits
  177         {
  178             blah = bitNums & mask;
  179             SER = (unsigned char)(blah >> (31-i));
  180             pulseSRCLK(); //shifts bit from SER --> QA, QA --> QB, etc
  181             mask = mask >> 1; //shifts bits in mask rightward by 1 bit
  182             //printf("SER out %lu\r\n", blah);
  183         }
  184     pulseRCLK(); //show bits on port
  185 }
