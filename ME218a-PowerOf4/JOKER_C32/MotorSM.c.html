   59  Parameters

   63      boolean, False if error in initialization, True otherwise

   66      Saves away the priority, sets up the initial transition and does any

   73 boolean InitMotorSM ( uint8_t Priority )
   74 {
   75     ES_Event ThisEvent;
   76     MyPriority = Priority;
   77     // initialize PWM library
   78     PWMS12_Init();
   79 
   80     // put us into stationary motor state
   81     CurrentState = Stationary;
   82     MOTOR_DIR = 1;  //set Port M0 as output for motor direction
   83 
   84     // post the initial transition event
   85     ThisEvent.EventType = ES_INIT;
   86     if (ES_PostToService( MyPriority, ThisEvent) == True)
   87         {
   88             return True;
   89         }
   90     else
   91         {
   92             return False;
   93         }
   94 }

  118 /****************************************************************************
  119  Function
  120     RunTemplateFSM
  121 
  122  Parameters
  123    ES_Event : the event to process
  124 
  125  Returns
  126    ES_Event, ES_NO_EVENT if no error ES_ERROR otherwise
  127 
  128  Description
  129    add your description here
  130  Notes
  131    uses nested switch/case to implement the machine.
  132  Author
  133    J. Edward Carryer, 01/15/12, 15:23
  134 ****************************************************************************/
  135 ES_Event RunMotorSM( ES_Event ThisEvent )
  136 {
  137     static MotorState_t NextState; //initiate NextState of motor
  138     ES_Event ReturnEvent;
  139     ReturnEvent.EventType = ES_NO_EVENT; // assume no errors
  140 
  141 
  142     CurrentState = QueryMotorSM();
  143 
  144     switch ( CurrentState )
  145         {
  146         case Stationary:
  147             if ( ThisEvent.EventType == MOTORSTART )
  148                 {
  149                     MotorSpeedDir(START);
  150                     puts("Motor Start detected bc Game started \r\n");
  151                     NextState = Moving;
  152                     //return ReturnEvent;
  153                 }
  154             if (ThisEvent.EventType == MOTORREVERSE)
  155                 {
  156                     puts("Reverse motor bc GAMEEND Game ended\r\n");
  157                     MotorSpeedDir(REVERSE);
  158                     NextState = Reverse;
  159 
  160                     //return ReturnEvent;
  161                 }
  162             if (ThisEvent.EventType == MOTORFAST)
  163                 {
  164                     MotorSpeedDir(FAST);
  165                     NextState = Moving;
  166                     //return ReturnEvent;
  167                 }
  168             break;
  169 
  170         case Moving:
  171             if ( ThisEvent.EventType == MOTORFAST )
  172                 {
  173                     MotorSpeedDir(FAST);
  174                     //puts("Motor Faster detected\r\n");
  175                     NextState = Moving;
  176                     //return ReturnEvent;
  177                 }
  178             if (ThisEvent.EventType == MOTORSLOW)
  179                 {
  180                     MotorSpeedDir(SLOW);
  181                     //puts("Motor Slower detected\r\n");
  182                     NextState = Moving;
  183                     //return ReturnEvent;
  184                 }
  185             if (ThisEvent.EventType == OILSLICK)
  186                 {
  187                     //puts("motor should be stopped\r\n");
  188                     MotorSpeedDir(STOP); //will need to modify to make it stop for 2 seconds
  189                     NextState = Stationary;
  190                     //return ReturnEvent;
  191                 }
  192             if (ThisEvent.EventType == MOTORSTOP)
  193                 {
  194                     MotorSpeedDir(STOP);
  195                     puts("Motor Stop detected\r\n");
  196                     NextState = Stationary;
  197                     //return ReturnEvent;
  198                 }
  199             break;
  200 
  201         case Reverse:
  202             if (ThisEvent.EventType == MOTORSTOP)
  203                 {
  204                     MotorSpeedDir(STOP);
  205                     puts("Motor Stopped\r\n");
  206                     NextState = Stationary;
  207                     //return ReturnEvent;
  208                 }
  209             break;
  210         }
  211     CurrentState = NextState;
  212     return ReturnEvent;
  213 }

  237 /***************************************************************************
  238  private functions
  239  ***************************************************************************/
  240 void MotorSpeedDir(unsigned char Command)
  241 {
  242     static unsigned char Speed[] = {25,30,35}; //duty cycle for PWM
  243     static unsigned char ind = 1; //default index of duty cycle = 50
  244 
  245     switch (Command)
  246         {
  247         case STOP:
  248             MOTORDIRECTION = 1; //default direction: T0 HIGH
  249             PWMS12_SetDuty(0, 0);
  250             break;
  251 
  252         case START:
  253             MOTORDIRECTION = 1; //default direction: T0 HIGH
  254             PWMS12_SetDuty(Speed[1], 0);
  255             break;
  256 
  257         case SLOW:
  258             MOTORDIRECTION = 1;
  259             if (ind <= 0)
  260                 {
  261                     ind = 0;
  262                     PWMS12_SetDuty(Speed[ind], 0);
  263                 }
  264             else
  265                 {
  266                     PWMS12_SetDuty(Speed[ind], 0);
  267                     ind--;
  268                 }
  269             break;
  270 
  271         case FAST:
  272             MOTORDIRECTION = 1;
  273             if (ind >= 2)
  274                 {
  275                     ind = 2;
  276                     PWMS12_SetDuty(Speed[ind], 0);
  277                 }
  278             else
  279                 {
  280                     PWMS12_SetDuty(Speed[ind], 0);
  281                     ind++;
  282                 }
  283             break;
  284 
  285         case REVERSE:
  286             MOTORDIRECTION = 0; //reverse direction T0 = LOW
  287             PWMS12_SetDuty(45, 0);
  288             break;
  289         }
  290 
  291     return;
  292 }
