   60      InitComSM
   61 
   62  Parameters
   63      uint8_t : the priorty of this service

   66      boolean, False if error in initialization, True otherwise
   67 
   68  Description
   69      Saves away the priority, sets up the initial transition and does any
   70      other required initialization for this state machine

   73  Author
   74      J. Edward Carryer, 10/23/11, 18:55

   76 boolean InitComSM ( uint8_t Priority )
   77 {
   78     ES_Event ThisEvent;
   79     MyPriority = Priority;
   80     printf("ComSM I AM C32-1\r\n");
   81 
   82     // put us into stationary motor state
   83     CurrentState = Idle;
   84 
   85 //set everything low before setting as inputs
   86     DDRT |= 0xF8; //sets ports 1-5 as outputs
   87 //write ports to 0
   88     STROBE = 0;
   89     D0 = 0;
   90     D1 = 0;
   91     D2 = 0;
   92     ACKNOWLEDGE = 0;
   93 // set ports as inputs
   94     STROBE_DIR = 0;
   95     D0_DIR = 0;
   96     D1_DIR = 0;
   97     D2_DIR = 0;
   98     ACK_DIR = 0;
   99 //printf("initialization complete \r\n");
  100 
  101     // post the initial transition event
  102     ThisEvent.EventType = ES_INIT;    //MAKE THIS ES_INIT FOR THE OTHER C32!!!!!!
  103     //isEvent.EventParam = GenerateNum(); //COMMENT THIS LINE OUT FOR THE OTHER C32
  104 
  105     if (ES_PostToService( MyPriority, ThisEvent) == True)
  106         {
  107             return True;
  108         }
  109     else
  110         {
  111             return False;
  112         }

  137 /****************************************************************************
  138  Function
  139     RunTemplateFSM
  140 
  141  Parameters
  142    ES_Event : the event to process
  143 
  144  Returns
  145    ES_Event, ES_NO_EVENT if no error ES_ERROR otherwise
  146 
  147  Description
  148    add your description here
  149  Notes
  150    uses nested switch/case to implement the machine.
  151  Author
  152    J. Edward Carryer, 01/15/12, 15:23
  153 ****************************************************************************/
  154 ES_Event RunComSM( ES_Event ThisEvent )
  155 {
  156     static ComState_t NextState; //initiate NextState of motor
  157 
  158     ES_Event ReturnEvent;
  159     ReturnEvent.EventType = ES_NO_EVENT; // assume no errors
  160 
  161     //CurrentState = QueryComSM();
  162     //NextState = CurrentState;
  163 
  164     if (ThisEvent.EventType == ES_INIT)
  165         {
  166             NextState = Idle;
  167         }
  168 
  169     switch ( CurrentState )
  170         {
  171         case Idle:
  172             if ((ThisEvent.EventType == SENDREQ) && (STROBE==1)) //guard: strobe asserted
  173                 {
  174                     NewData = ThisEvent.EventParam; //Param of SENDREQ is the info of the 3bits save it!!
  175                     //printf("IDLE --> WAIT2RECEIVE. newdata = %i\r\n", NewData);
  176                     NextState = Wait2Receive;
  177                 }
  178 
  179             if ((ThisEvent.EventType == SENDREQ) && (STROBE==0)) //guard: strobe deasserted
  180                 {
  181                     printf("SENDREQ --> WAIT4ACK \r\n");
  182 
  183 
  184                     STROBE_DIR = 1; // set strobe as output
  185                     STROBE = 1; //set strobe high
  186 
  187                     D0_DIR = 1; //assert data as output
  188                     D1_DIR = 1;
  189                     D2_DIR = 1;
  190 
  191                     if ((ThisEvent.EventParam & BIT0HI) != 0)
  192                         {
  193                             D0 = 1; //assert Data0 to T4
  194                         }
  195                     else
  196                         {
  197                             D0 = 0;
  198                         }
  199                     if ((ThisEvent.EventParam & BIT1HI) != 0)
  200                         {
  201                             D1 = 1; //assert Data1 to T5
  202                         }
  203                     else
  204                         {
  205                             D1 = 0;
  206                         }
  207                     if ((ThisEvent.EventParam & BIT2HI) != 0)
  208                         {
  209                             D2 = 1; //assert Data2 to T6
  210                         }
  211                     else
  212                         {
  213                             D2 = 0;
  214                         }
  215 
  216                     Data[0] = D2;
  217                     Data[1] = D1;
  218                     Data[2] = D0;//read data into Data  [MSB->LSB]
  219                     Message = ((Data[2])|(Data[1]<<1)|(Data[0]<<2));
  220 
  221                     ES_Timer_InitTimer(COM_TIMER, TIMERVAL); //start timer
  222                     NextState = Wait4Ack;
  223                     //return ReturnEvent;
  224                 }
  225 
  226             if (ThisEvent.EventType == STROBE_RISE)
  227                 {
  228 
  229                     ES_Event ThisEvent;
  230                     Data[0] = D2;
  231                     Data[1] = D1;
  232                     Data[2] = D0;//read data into Data  [MSB->LSB]
  233                     Message = ((Data[2])|(Data[1]<<1)|(Data[0]<<2));
  234 
  235                     //Message = ((D0)|(D1<<1)|(D2<<2)); //bitshift D1 left 1, bitshift D2 left 2 times to get (D2)(D1)(D0)
  236 
  237                     printf("I got Data %i \r\n", Message);
  238 
  239 
  240                     ThisEvent.EventType = InterpretMsg(Message);
  241                     ThisEvent.EventParam = FROMBATMAN;
  242                     PostGameplaySM(ThisEvent);
  243                     PostTimerSM(ThisEvent);
  244 
  245                     ACK_DIR = 1; //set ACK as output
  246                     ACKNOWLEDGE = 1; //set ACK high
  247                     NextState = WaitStrobeFall;
  248                     //return ReturnEvent;
  249                 }
  250             break;
  251 
  252         case WaitStrobeFall:
  253             if (ThisEvent.EventType == STROBE_FALL)
  254                 {
  255                     //printf("WAITSTROBEFALL --> IDLE\r\n");
  256 
  257                     ACKNOWLEDGE = 0; //bring ACK low
  258                     ACK_DIR = 0; //deassert ACK
  259                     NextState = Idle;
  260                     //return ReturnEvent;
  261                 }
  262             break;
  263 
  264         case Wait2Receive:
  265             printf("Wait2Receive \r\n");
  266             if (ThisEvent.EventType == STROBE_RISE)
  267                 {
  268                     ES_Event ThisEvent;
  269                     Data[0] = D2;
  270                     Data[1] = D1;
  271                     Data[2] = D0;//read data into Data  [MSB->LSB]
  272                     Message = ((Data[2])|(Data[1]<<1)|(Data[0]<<2));
  273                     //Message = ((D0)|(D1<<1)|(D2<<2)); //bitshift D1 left 1, bitshift D2 left 2 times to get (D2)(D1)(D0)
  274 
  275                     printf("I got Data %i and posting to Gameplay\r\n", Message);
  276 
  277 
  278                     ThisEvent.EventType = InterpretMsg(Message);
  279                     ThisEvent.EventParam = FROMBATMAN;
  280                     PostGameplaySM(ThisEvent);
  281 
  282                     ACK_DIR = 1; //assert ACK
  283                     ACKNOWLEDGE = 1; //bring ACK high
  284                     NextState = Wait2Send;
  285                     //return ReturnEvent;
  286                 }
  287             break;
  288 
  289         case Wait2Send:
  290             printf("Wait2Send \r\n");
  291             if (ThisEvent.EventType == STROBE_FALL)
  292                 {
  293                     //printf("I'm in wait2send, asserting strobe and data, going to wait4ack \r\n");
  294                     ACKNOWLEDGE = 0; //deassert ACKKNOWLEDGE
  295                     ACK_DIR = 0; //set ACK as input
  296 
  297                     STROBE_DIR = 1; //set strobe as output
  298                     STROBE = 1; //bring strobe high
  299 
  300 
  301                     D0_DIR = 1; //assert NewData
  302                     D1_DIR = 1;
  303                     D2_DIR = 1;
  304 
  305 
  306                     if ((NewData & BIT0HI) != 0)
  307                         {
  308                             D0 = 1; //assert Data0
  309                         }
  310                     else
  311                         {
  312                             D0 = 0;
  313                         }
  314                     if ((NewData & BIT1HI) != 0)
  315                         {
  316                             D1 = 1; //assert Data1
  317                         }
  318                     else
  319                         {
  320                             D1 = 0;
  321                         }
  322                     if ((NewData & BIT2HI) != 0)
  323                         {
  324                             D2 = 1; //assert Data2
  325                         }
  326                     else
  327                         {
  328                             D2 = 0;
  329                         }
  330 
  331                     ES_Timer_InitTimer(COM_TIMER, TIMERVAL);
  332                     NextState = Wait4Ack;
  333                     //return ReturnEvent;
  334                 }
  335             break;
  336 
  337         case Wait4Ack:
  338             //printf("Wait4Ack \r\n");
  339             if (ThisEvent.EventType == ACK_RISE)
  340                 {
  341                     printf("other person got the data %i\r\n", Message);
  342                     STROBE_DIR = 1; //output
  343                     STROBE = 0; //strobe low
  344                     STROBE_DIR = 0; //input
  345 
  346 
  347 
  348                     //reset data pins
  349                     D0 = 0;
  350                     D1 = 0;
  351                     D2 = 0;
  352                     //deassert data pins
  353                     D0_DIR = 0; //deassert data0
  354                     D1_DIR = 0;  //deassert data1
  355                     D2_DIR = 0;//deassert data2
  356 
  357                     NextState = Idle;
  358                     //return ReturnEvent;
  359                 }
  360 
  361             if ((ThisEvent.EventType == ES_TIMEOUT) && (ThisEvent.EventParam == COM_TIMER))
  362                 {
  363                     //printf("resend data, also deasserting strobe \r\n");
  364                     STROBE_DIR = 1; //output
  365                     STROBE = 0; //strobe low
  366                     STROBE_DIR = 0; //input
  367 
  368 
  369                     ThisEvent.EventType = SENDREQ;
  370                     ThisEvent.EventParam = Message;  //resend message that was posted
  371                     PostComSM(ThisEvent);
  372 
  373                     NextState = Idle;
  374                     //return ReturnEvent;
  375                 }
  376             break;
  377         }
  378     CurrentState = NextState;
  379     return ReturnEvent;
  380 }
